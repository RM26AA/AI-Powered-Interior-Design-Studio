import React from "react";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Heart, Download, Share2, RotateCcw } from "lucide-react";
import { motion } from "framer-motion";

export default function StyledResult({ 
  originalImage, 
  styledImage, 
  styleName, 
  onSave, 
  onTryAgain, 
  onShare,
  isSaving 
}) {
  const handleDownload = () => {
    const link = document.createElement('a');
    link.href = styledImage;
    link.download = `styled-room-${styleName}-${Date.now()}.jpg`;
    link.click();
  };

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      className="space-y-6"
    >
      <div className="text-center">
        <h3 className="text-2xl font-bold text-gray-900 mb-2">Your Styled Room</h3>
        <Badge className="bg-yellow-100 text-yellow-800 px-4 py-2 rounded-full">
          {styleName.charAt(0).toUpperCase() + styleName.slice(1)} Style
        </Badge>
      </div>

      <div className="grid md:grid-cols-2 gap-6">
        <Card className="overflow-hidden shadow-lg">
          <CardContent className="p-0">
            <div className="relative">
              <img
                src={typeof originalImage === 'string' ? originalImage : URL.createObjectURL(originalImage)}
                alt="Original room"
                className="w-full h-64 object-cover"
              />
              <div className="absolute bottom-4 left-4">
                <Badge className="bg-black/80 text-white">Original</Badge>
              </div>
            </div>
          </CardContent>
        </Card>

        <Card className="overflow-hidden shadow-lg ring-2 ring-yellow-400">
          <CardContent className="p-0">
            <div className="relative">
              <img
                src={styledImage}
                alt="Styled room"
                className="w-full h-64 object-cover"
              />
              <div className="absolute bottom-4 left-4">
                <Badge className="bg-yellow-500 text-white">Styled</Badge>
              </div>
            </div>
          </CardContent>
        </Card>
      </div>

      <div className="flex flex-wrap gap-3 justify-center">
        <Button
          onClick={onSave}
          disabled={isSaving}
          className="bg-green-600 hover:bg-green-700 text-white px-6 py-3 rounded-xl font-medium shadow-lg hover:shadow-xl transition-all duration-300"
        >
          <Heart className="w-5 h-5 mr-2" />
          {isSaving ? 'Saving...' : 'Save to Gallery'}
        </Button>

        <Button
          onClick={handleDownload}
          variant="outline"
          className="px-6 py-3 rounded-xl font-medium border-gray-300 hover:bg-gray-50 transition-all duration-300"
        >
          <Download className="w-5 h-5 mr-2" />
          Download
        </Button>

        <Button
          onClick={onShare}
          variant="outline"
          className="px-6 py-3 rounded-xl font-medium border-gray-300 hover:bg-gray-50 transition-all duration-300"
        >
          <Share2 className="w-5 h-5 mr-2" />
          Share
        </Button>

        <Button
          onClick={onTryAgain}
          variant="outline"
          className="px-6 py-3 rounded-xl font-medium border-yellow-400 text-yellow-700 hover:bg-yellow-50 transition-all duration-300"
        >
          <RotateCcw className="w-5 h-5 mr-2" />
          Try Another Style
        </Button>
      </div>
    </motion.div>
  );
}
